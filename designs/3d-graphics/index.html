scene.add(pointLight); function animatePrism() { prism.rotation.x += 0.01; prism.rotation.y += 0.02; prism.position.y = Math.sin(Date.now() * 0.001) * 0.5; renderer.render(scene, camera); requestAnimationFrame(animatePrism); const camera = new THREE.PerspectiveCamera(75, parentWidth / 280, 0.1, 1000); const renderer = new THREE.WebGLRenderer({ canvas, alpha: true, antialias: true }); renderer.setSize(parentWidth, 280); renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2)); const particleSystem = new THREE.Points(particles, material); scene.add(particleSystem); function animateParticles() { particleSystem.rotation.y += 0.005; particleSystem.rotation.x += 0.002; const positions = particleSystem.geometry.attributes.position.array; for (let i = 0; i < particleCount; i++) { const angle = Date.now() * 0.001 + i * 0.1; positions[i * 3] = Math.cos(angle) * (2 + Math.sin(angle * 0.5)); positions[i * 3 + 1] = Math.sin(angle * 0.7) * 2; positions[i * 3 + 2] = Math.sin(angle) * (2 + Math.cos(angle * 0.5)); } particleSystem.geometry.attributes.position.needsUpdate = true; renderer.render(scene, camera); requestAnimationFrame(animateParticles);
